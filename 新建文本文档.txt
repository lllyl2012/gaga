动漫电影
create table animation(
id int auto_increment primary key comment '字段的注释',
name varchar(50) comment '字段的注释',
director varchar(30) comment '字段的注释',
create_time date comment '字段的注释',
language varchar(10) comment '字段的注释',
resolution_ratio varchar(5) comment '字段的注释',
format varchar(10) comment '字段的注释',
conntry varchar(20) comment '字段的注释',
subtitles varchar(5) comment '字段的注释',
bean_score varchar(5) comment '字段的注释',
size varchar(10) comment '字段的注释',
time varchar(10) comment '字段的注释',
remark varchar(255) comment '字段的注释'
)comment='表的注释';

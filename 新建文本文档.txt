动漫电影
create table animation(
id int auto_increment primary key,
name varchar(50),
director varchar(30),
create_time date,
language varchar(10),
resolution_ratio varchar(5),
format varchar(10),
conntry varchar(20),
subtitles varchar(5),
bean_score varchar(5),
size varchar(10),
time varchar(10),
remark varchar(255)
);
